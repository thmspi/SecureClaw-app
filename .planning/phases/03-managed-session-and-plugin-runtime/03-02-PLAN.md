```typescript
const MAX_CONCURRENT_PLUGINS = 3; // Agent discretion value

class PluginRunner {
  private queue: PluginRunRequest[] = [];
  private active: Map<string, PluginRunState> = new Map();

  async enqueue(request: PluginRunRequest): Promise<void> {
    this.queue.push(request);
    this.emitState(request.runId, 'Queued');
    this.processQueue();
  }

  private async processQueue(): Promise<void> {
    while (this.queue.length > 0 && this.active.size < MAX_CONCURRENT_PLUGINS) {
      const request = this.queue.shift()!;
      this.active.set(request.runId, { state: 'Starting' });
      this.executePlugin(request);
    }
  }
}
```

**Implementation structure:**

```typescript
import { runProcess, stopProcess } from './process-runner';
import { getSessions } from './session-orchestrator';
import type {
  PluginRun,
  PluginRunState,
  PluginEvent,
  PluginEventType,
  RunPluginRequest,
  RunPluginResponse,
  CancelPluginRequest,
  CancelPluginResponse,
  GetPluginRunsRequest,
  GetPluginRunsResponse,
} from '../../shared/runtime/runtime-contracts';

const MAX_CONCURRENT_PLUGINS = 3;  // D-08 agent discretion

// Plugin run registry
const pluginRuns = new Map<string, PluginRun>();
const pluginQueue: RunPluginRequest[] = [];
const activePlugins = new Set<string>();
const pluginEventListeners = new Map<string, (event: PluginEvent) => void>();

function emitPluginEvent(
  runId: string,
  type: PluginEventType,
  data: Record<string, unknown> = {}
): void {
  const event: PluginEvent = {
    type,
    runId,
    timestamp: new Date().toISOString(),
    data,
  };
  const listener = pluginEventListeners.get(runId);
  listener?.(event);
}

export function registerPluginEventListener(
  runId: string,
  listener: (event: PluginEvent) => void
): void {
  pluginEventListeners.set(runId, listener);
}

export function unregisterPluginEventListener(runId: string): void {
  pluginEventListeners.delete(runId);
}

function processQueue(): void {
  while (pluginQueue.length > 0 && activePlugins.size < MAX_CONCURRENT_PLUGINS) {
    const request = pluginQueue.shift()!;
    void executePlugin(request);
  }
}

async function executePlugin(request: RunPluginRequest): Promise<void> {
  const { runId, pluginName, sessionId, args } = request;
  const run = pluginRuns.get(runId);
  if (!run) return;

  activePlugins.add(runId);

  try {
    // Transition Queued → Starting
    run.state = 'Starting';
    pluginRuns.set(runId, { ...run });
    emitPluginEvent(runId, 'starting');

    // Transition Starting → Running
    run.state = 'Running';
    run.startedAt = new Date().toISOString();
    pluginRuns.set(runId, { ...run });
    emitPluginEvent(runId, 'running');

    // Execute plugin command via process runner
    await runProcess({
      command: 'openclaw',
      args: ['plugin', 'run', pluginName, ...(args ? [JSON.stringify(args)] : [])],
      correlationId: runId,
      onEvent: (evt) => {
        // Capture stdout/stderr as logs
        if (evt.type === 'stdout' || evt.type === 'stderr') {
          run.logs.push(evt.data.chunk as string);
          pluginRuns.set(runId, { ...run });
        }
      },
    });

    // Success - transition Running → Completed
    run.state = 'Completed';
    run.completedAt = new Date().toISOString();
    if (run.startedAt) {
      run.durationMs = new Date(run.completedAt).getTime() - new Date(run.startedAt).getTime();
    }
    pluginRuns.set(runId, { ...run });
    emitPluginEvent(runId, 'completed');
  } catch (error) {
    // Failure - transition to Failed
    run.state = 'Failed';
    run.completedAt = new Date().toISOString();
    run.error = error instanceof Error ? error.message : String(error);
    if (run.startedAt && run.completedAt) {
      run.durationMs = new Date(run.completedAt).getTime() - new Date(run.startedAt).getTime();
    }
    pluginRuns.set(runId, { ...run });
    emitPluginEvent(runId, 'failed', { error: run.error });
  } finally {
    // Always clean up and process next in queue
    activePlugins.delete(runId);
    processQueue();
  }
}

export async function enqueuePlugin(
  request: RunPluginRequest
): Promise<RunPluginResponse> {
  const { runId, pluginName, sessionId } = request;

  // D-07: Validate session is Active
  const sessions = getSessions();
  const session = sessions.find(s => s.sessionId === sessionId);
  if (!session || session.state !== 'Active') {
    return {
      runId,
      queued: false,
      error: 'Session not active - start a session before running plugins',
    };
  }

  // Initialize plugin run
  const run: PluginRun = {
    runId,
    pluginName,
    sessionId,
    state: 'Queued',
    queuedAt: new Date().toISOString(),
    logs: [],
  };
  pluginRuns.set(runId, run);
  emitPluginEvent(runId, 'queued');

  // Add to queue and attempt immediate execution
  pluginQueue.push(request);
  processQueue();

  return { runId, queued: true };
}

export async function cancelPluginRun(
  request: CancelPluginRequest
): Promise<CancelPluginResponse> {
  const { runId } = request;
  const run = pluginRuns.get(runId);

  if (!run) {
    return { runId, cancelled: false, error: 'Plugin run not found' };
  }

  if (run.state === 'Completed' || run.state === 'Failed') {
    return { runId, cancelled: false, error: 'Plugin run already finished' };
  }

  try {
    // If running, stop the process
    if (run.state === 'Running' || run.state === 'Starting') {
      await stopProcess(runId, { strategy: 'force' });
    }

    // Mark as Failed with cancelled metadata
    run.state = 'Failed';
    run.completedAt = new Date().toISOString();
    run.error = 'Cancelled by user';
    pluginRuns.set(runId, { ...run });
    emitPluginEvent(runId, 'cancelled');

    activePlugins.delete(runId);
    processQueue();

    return { runId, cancelled: true };
  } catch (error) {
    return {
      runId,
      cancelled: false,
      error: error instanceof Error ? error.message : String(error),
    };
  }
}

export function getPluginRuns(request: GetPluginRunsRequest): GetPluginRunsResponse {
  let runs = Array.from(pluginRuns.values());

  // Filter by sessionId if provided
  if (request.sessionId) {
    runs = runs.filter(r => r.sessionId === request.sessionId);
  }

  return { runs };
}
```
