Create session-orchestrator.ts that manages session lifecycle with explicit state machine validation.

**State Machine (per D-03, D-05, D-06):**
- Idle → Starting (on startSession)
- Starting → Active (on health probe success)
- Starting → Stopped (on health probe timeout or spawn failure)
- Active → Stopping (on stopSession)
- Stopping → Stopped (on process exit)

**Implementation structure:**

```typescript
import { runProcess, stopProcess } from './process-runner';
import type {
  ManagedSession,
  SessionState,
  SessionEvent,
  StartSessionRequest,
  StartSessionResponse,
  StopSessionRequest,
  StopSessionResponse,
} from '../../shared/runtime/runtime-contracts';

// Session registry - supports multiple concurrent sessions (D-03)
const activeSessions = new Map<string, ManagedSession>();
const sessionEventListeners = new Map<string, (event: SessionEvent) => void>();

// Valid state transitions
const VALID_TRANSITIONS: Record<SessionState, SessionState[]> = {
  'Idle': ['Starting'],
  'Starting': ['Active', 'Stopped'],
  'Active': ['Stopping'],
  'Stopping': ['Stopped'],
  'Stopped': ['Idle'],
};

function validateTransition(from: SessionState, to: SessionState): boolean {
  return VALID_TRANSITIONS[from]?.includes(to) ?? false;
}

function emitSessionEvent(
  sessionId: string,
  type: SessionEventType,
  data: Record<string, unknown> = {}
): void {
  const event: SessionEvent = {
    type,
    sessionId,
    timestamp: new Date().toISOString(),
    data,
  };
  const listener = sessionEventListeners.get(sessionId);
  listener?.(event);
}

// Health probe for readiness confirmation (D-05)
async function waitForReadiness(
  healthEndpoint: string,
  timeoutMs: number = 30000
): Promise<boolean> {
  const startTime = Date.now();
  const pollIntervalMs = 500;

  while (Date.now() - startTime < timeoutMs) {
    try {
      const res = await fetch(healthEndpoint, { method: 'GET', signal: AbortSignal.timeout(2000) });
      if (res.ok) {
        return true; // Ready!
      }
    } catch {
      // Not ready yet, continue polling
    }
    await new Promise(resolve => setTimeout(resolve, pollIntervalMs));
  }
  return false; // Timeout
}

// Lightweight preflight checks (D-04)
async function runSessionPreflight(): Promise<{ passed: boolean; message?: string }> {
  // Placeholder - could check Docker daemon, disk space, etc.
  // For Phase 3, assume pass. Phase 4 may expand.
  return { passed: true };
}

export function registerSessionEventListener(
  sessionId: string,
  listener: (event: SessionEvent) => void
): void {
  sessionEventListeners.set(sessionId, listener);
}

export function unregisterSessionEventListener(sessionId: string): void {
  sessionEventListeners.delete(sessionId);
}

export async function startSession(
  request: StartSessionRequest
): Promise<StartSessionResponse> {
  const { sessionId, config } = request;
  const healthEndpoint = config?.healthEndpoint ?? 'http://localhost:8080/health';
  const readinessTimeoutMs = config?.readinessTimeoutMs ?? 30000;

  // D-04: Lightweight preflight checks
  const preflight = await runSessionPreflight();
  if (!preflight.passed) {
    return { sessionId, started: false, error: preflight.message };
  }

  // Initialize session in Starting state
  const session: ManagedSession = {
    sessionId,
    state: 'Starting',
    startedAt: new Date().toISOString(),
  };
  activeSessions.set(sessionId, session);
  emitSessionEvent(sessionId, 'starting');

  try {
    // Spawn OpenClaw process
    await runProcess({
      command: 'openclaw',
      args: ['serve'],
      correlationId: sessionId,
      onEvent: (evt) => {
        if (evt.type === 'spawned') {
          session.pid = evt.data.pid as number;
          activeSessions.set(sessionId, { ...session });
        }
      },
    });

    // D-05: Wait for readiness confirmation before marking Active
    const ready = await waitForReadiness(healthEndpoint, readinessTimeoutMs);
    
    if (ready) {
      // Transition Starting → Active
      session.state = 'Active';
      session.activeAt = new Date().toISOString();
      activeSessions.set(sessionId, { ...session });
      emitSessionEvent(sessionId, 'active');
      return { sessionId, started: true };
    } else {
      // Timeout - transition Starting → Stopped
      session.state = 'Stopped';
      session.error = `Readiness timeout after ${readinessTimeoutMs}ms`;
      activeSessions.set(sessionId, { ...session });
      emitSessionEvent(sessionId, 'error', { error: session.error });
      return { sessionId, started: false, error: session.error };
    }
  } catch (error) {
    session.state = 'Stopped';
    session.error = error instanceof Error ? error.message : String(error);
    activeSessions.set(sessionId, { ...session });
    emitSessionEvent(sessionId, 'error', { error: session.error });
    return { sessionId, started: false, error: session.error };
  }
}

export async function stopSession(
  request: StopSessionRequest
): Promise<StopSessionResponse> {
  const { sessionId, strategy = 'graceful' } = request;
  const session = activeSessions.get(sessionId);

  if (!session || session.state !== 'Active') {
    return { sessionId, stopped: false, error: 'Session not active' };
  }

  // Transition Active → Stopping
  session.state = 'Stopping';
  session.stoppedAt = new Date().toISOString();
  activeSessions.set(sessionId, { ...session });
  emitSessionEvent(sessionId, 'stopping', { strategy });

  try {
    // D-06: Use existing graceful-to-force escalation from process-runner
    await stopProcess(sessionId, { strategy });

    // Transition Stopping → Stopped
    session.state = 'Stopped';
    activeSessions.set(sessionId, { ...session });
    emitSessionEvent(sessionId, 'stopped');
    return { sessionId, stopped: true };
  } catch (error) {
    session.error = error instanceof Error ? error.message : String(error);
    activeSessions.set(sessionId, { ...session });
    return { sessionId, stopped: false, error: session.error };
  }
}

export function getSessions(): ManagedSession[] {
  return Array.from(activeSessions.values());
}
```

**Implementation structure:**

```typescript
import Database from 'better-sqlite3';
import { app } from 'electron';
import { join } from 'path';
import type {
  RuntimeHistoryRecord,
  GetHistoryRequest,
  GetHistoryResponse,
  OperationType,
  OperationStatus,
} from '../../shared/runtime/runtime-contracts';

const DB_PATH = join(app.getPath('userData'), 'secureclaw.db');
const RETENTION_DAYS = 90;  // D-17

let db: Database.Database | null = null;

function getDb(): Database.Database {
  if (!db) {
    db = new Database(DB_PATH);
    db.pragma('journal_mode = WAL');
    
    // Create runtime_history table
    db.exec(`
      CREATE TABLE IF NOT EXISTS runtime_history (
        id TEXT PRIMARY KEY,
        operation_type TEXT NOT NULL,
        status TEXT NOT NULL,
        target_name TEXT,
        started_at TEXT NOT NULL,
        completed_at TEXT,
        duration_ms INTEGER,
        error_message TEXT,
        error_details TEXT,
        metadata TEXT
      )
    `);
    
    // Index for filtering by type and date (D-18)
    db.exec(`
      CREATE INDEX IF NOT EXISTS idx_history_type_date 
        ON runtime_history(operation_type, started_at)
    `);
    
    db.exec(`
      CREATE INDEX IF NOT EXISTS idx_history_status
        ON runtime_history(status)
    `);
  }
  return db;
}

export function saveHistoryRecord(record: RuntimeHistoryRecord): void {
  const stmt = getDb().prepare(`
    INSERT INTO runtime_history (
      id, operation_type, status, target_name,
      started_at, completed_at, duration_ms,
      error_message, error_details, metadata
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)
  `);
  
  stmt.run(
    record.id,
    record.operationType,
    record.status,
    record.targetName ?? null,
    record.startedAt,
    record.completedAt ?? null,
    record.durationMs ?? null,
    record.errorMessage ?? null,
    record.errorDetails ?? null,
    record.metadata ? JSON.stringify(record.metadata) : null
  );
}

export function getHistory(request: GetHistoryRequest): GetHistoryResponse {
  let query = 'SELECT * FROM runtime_history WHERE 1=1';
  const params: unknown[] = [];

  // D-18: Apply filters
  if (request.operationType) {
    query += ' AND operation_type = ?';
    params.push(request.operationType);
  }

  if (request.status) {
    query += ' AND status = ?';
    params.push(request.status);
  }

  if (request.fromDate) {
    query += ' AND started_at >= ?';
    params.push(request.fromDate);
  }

  if (request.toDate) {
    query += ' AND started_at <= ?';
    params.push(request.toDate);
  }

  query += ' ORDER BY started_at DESC';

  if (request.limit) {
    query += ' LIMIT ?';
    params.push(request.limit);
  }

  const rows = getDb().prepare(query).all(...params) as Array<Record<string, unknown>>;

  const records: RuntimeHistoryRecord[] = rows.map(row => ({
    id: row.id as string,
    operationType: row.operation_type as OperationType,
    status: row.status as OperationStatus,
    targetName: row.target_name as string | undefined,
    startedAt: row.started_at as string,
    completedAt: row.completed_at as string | undefined,
    durationMs: row.duration_ms as number | undefined,
    errorMessage: row.error_message as string | undefined,
    errorDetails: row.error_details as string | undefined,
    metadata: row.metadata ? JSON.parse(row.metadata as string) : undefined,
  }));

  return {
    records,
    total: records.length,
  };
}

export function cleanupOldRecords(): { deleted: number } {
  const cutoffDate = new Date();
  cutoffDate.setDate(cutoffDate.getDate() - RETENTION_DAYS);
  
  const stmt = getDb().prepare('DELETE FROM runtime_history WHERE started_at < ?');
  const result = stmt.run(cutoffDate.toISOString());
  
  return { deleted: result.changes };
}

// For testing
export function _setDbForTesting(testDb: Database.Database): void {
  db = testDb;
}

export function _resetDbForTesting(): void {
  db = null;
}
```
