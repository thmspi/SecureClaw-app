import { ipcMain, IpcMainInvokeEvent, BrowserWindow } from 'electron';
import {
  RUNTIME_CHANNELS,
  startSessionSchema,
  stopSessionSchema,
  getSessionsSchema,

  cancelPluginSchema,

import type {

  CancelPluginRequest,

  GetHistoryRequest,
  GetHistoryResponse,
  SessionEvent,
  PluginEvent,

import * as pluginRunner from '../runtime/plugin-runner';
import * as historyService from '../runtime/runtime-history-service';

let mainWindow: BrowserWindow | null = null;

export function setMainWindow(window: BrowserWindow): void {
  mainWindow = window;
}

// Forward session events to renderer
function forwardSessionEvent(event: SessionEvent): void {
  if (mainWindow) {
    mainWindow.webContents.send(RUNTIME_CHANNELS.sessionEvent, event);
  }
}

// Forward plugin events to renderer
function forwardPluginEvent(event: PluginEvent): void {
  if (mainWindow) {
    mainWindow.webContents.send(RUNTIME_CHANNELS.pluginEvent, event);
  }
}

export function registerRuntimeHandlers(ipc: typeof ipcMain): void {
  // Session handlers
  ipc.handle(
    RUNTIME_CHANNELS.startSession,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<StartSessionResponse> => {
      const validated = startSessionSchema.parse(request) as StartSessionRequest;
      
      // Register event listener to forward to renderer
      sessionOrchestrator.registerSessionEventListener(validated.sessionId, forwardSessionEvent);
      
      const response = await sessionOrchestrator.startSession(validated);
      
      // Save to history
      if (response.started) {
        historyService.saveHistoryRecord({
          id: validated.sessionId,
          operationType: 'session_start',
          status: 'success',
          targetName: validated.sessionId,
          startedAt: new Date().toISOString(),
          completedAt: new Date().toISOString(),
          durationMs: 0,
        });
      } else {
        historyService.saveHistoryRecord({
          id: validated.sessionId,
          operationType: 'session_start',
          status: 'failed',
          targetName: validated.sessionId,
          startedAt: new Date().toISOString(),
          completedAt: new Date().toISOString(),
          errorMessage: response.error,
        });
      }
      
      return response;
    }
  );

  ipc.handle(
    RUNTIME_CHANNELS.stopSession,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<StopSessionResponse> => {
      const validated = stopSessionSchema.parse(request) as StopSessionRequest;
      const response = await sessionOrchestrator.stopSession(validated);
      
      // Save to history
      historyService.saveHistoryRecord({
        id: crypto.randomUUID(),
        operationType: 'session_stop',
        status: response.stopped ? 'success' : 'failed',
        targetName: validated.sessionId,
        startedAt: new Date().toISOString(),
        completedAt: new Date().toISOString(),
        errorMessage: response.error,
      });
      
      // Unregister event listener
      sessionOrchestrator.unregisterSessionEventListener(validated.sessionId);
      
      return response;
    }
  );

  ipc.handle(
    RUNTIME_CHANNELS.getSessions,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<GetSessionsResponse> => {
      getSessionsSchema.parse(request);
      const sessions = sessionOrchestrator.getSessions();
      return { sessions };
    }
  );

  // Plugin handlers
  ipc.handle(
    RUNTIME_CHANNELS.runPlugin,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<RunPluginResponse> => {
      const validated = runPluginSchema.parse(request) as RunPluginRequest;
      
      // Register event listener to forward to renderer
      pluginRunner.registerPluginEventListener(validated.runId, forwardPluginEvent);
      
      const response = await pluginRunner.enqueuePlugin(validated);
      
      // History will be saved on plugin completion via event listener
      
      return response;
    }
  );

  ipc.handle(
    RUNTIME_CHANNELS.cancelPluginRun,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<CancelPluginResponse> => {
      const validated = cancelPluginSchema.parse(request) as CancelPluginRequest;
      const response = await pluginRunner.cancelPluginRun(validated);
      
      // Unregister event listener
      if (response.cancelled) {
        pluginRunner.unregisterPluginEventListener(validated.runId);
      }
      
      return response;
    }
  );

  ipc.handle(
    RUNTIME_CHANNELS.getPluginRuns,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<GetPluginRunsResponse> => {
      const validated = getPluginRunsSchema.parse(request) as GetPluginRunsRequest;
      return pluginRunner.getPluginRuns(validated);
    }
  );

  // History handler
  ipc.handle(
    RUNTIME_CHANNELS.getHistory,
    async (_event: IpcMainInvokeEvent, request: unknown): Promise<GetHistoryResponse> => {
      const validated = getHistorySchema.parse(request) as GetHistoryRequest;
      return historyService.getHistory(validated);
    }
  );
}

// Plugin event listener to save to history on completion
function savePluginToHistory(event: PluginEvent): void {
  if (event.type === 'completed' || event.type === 'failed') {
    const runs = pluginRunner.getPluginRuns({});
    const run = runs.runs.find(r => r.runId === event.runId);
    if (run) {
      historyService.saveHistoryRecord({
        id: run.runId,
        operationType: 'plugin_run',
        status: event.type === 'completed' ? 'success' : 'failed',
        targetName: run.pluginName,
        startedAt: run.startedAt ?? new Date().toISOString(),
        completedAt: run.completedAt,
        durationMs: run.durationMs,
        errorMessage: run.error,
      });
    }
  }
}

```typescript
import { create } from 'zustand';
import type {
  ManagedSession,
  PluginRun,
  RuntimeHistoryRecord,
  GetHistoryRequest,
  SessionEvent,
  PluginEvent,
} from '../../shared/runtime/runtime-contracts';

interface ManagementState {
  // Session state
  sessions: ManagedSession[];
  activeSessions: ManagedSession[];
  
  // Plugin state
  pluginRuns: PluginRun[];
  
  // History state
  history: RuntimeHistoryRecord[];
  historyFilters: {
    operationType?: 'session_start' | 'session_stop' | 'plugin_run';
    status?: 'success' | 'failed' | 'cancelled';
    fromDate?: string;
    toDate?: string;
  };
  historyLoading: boolean;
  
  // UI state
  sessionStarting: boolean;
  sessionStopping: boolean;

  // Actions - Session
  loadSessions: () => Promise<void>;
  startSession: (sessionId: string, healthEndpoint?: string) => Promise<{ started: boolean; error?: string }>;
  stopSession: (sessionId: string) => Promise<{ stopped: boolean; error?: string }>;
  handleSessionEvent: (event: SessionEvent) => void;

  // Actions - Plugin
  loadPluginRuns: (sessionId?: string) => Promise<void>;
  runPlugin: (runId: string, pluginName: string, sessionId: string, args?: Record<string, unknown>) => Promise<{ queued: boolean; error?: string }>;
  cancelPluginRun: (runId: string) => Promise<{ cancelled: boolean; error?: string }>;
  handlePluginEvent: (event: PluginEvent) => void;

  // Actions - History
  loadHistory: () => Promise<void>;
  setHistoryFilters: (filters: Partial<GetHistoryRequest>) => void;
  applyHistoryFilters: () => Promise<void>;
}

export const useManagementStore = create<ManagementState>((set, get) => ({
  // Initial state
  sessions: [],
  activeSessions: [],
  pluginRuns: [],
  history: [],
  historyFilters: {},
  historyLoading: false,
  sessionStarting: false,
  sessionStopping: false,

  // Session actions
  loadSessions: async () => {
    try {
      const response = await window.secureClaw.runtime.getSessions();
      set({
        sessions: response.sessions,
        activeSessions: response.sessions.filter(s => s.state === 'Active'),
      });
    } catch (error) {
      console.error('Failed to load sessions:', error);
    }
  },

  startSession: async (sessionId: string, healthEndpoint?: string) => {
    set({ sessionStarting: true });
    try {
      const response = await window.secureClaw.runtime.startSession({
        sessionId,
        config: healthEndpoint ? { healthEndpoint } : undefined,
      });
      
      if (response.started) {
        await get().loadSessions();
      }
      
      return { started: response.started, error: response.error };
    } catch (error) {
      return { started: false, error: String(error) };
    } finally {
      set({ sessionStarting: false });
    }
  },

  stopSession: async (sessionId: string) => {
    set({ sessionStopping: true });
    try {
      const response = await window.secureClaw.runtime.stopSession({ sessionId });
      
      if (response.stopped) {
        await get().loadSessions();
      }
      
      return { stopped: response.stopped, error: response.error };
    } catch (error) {
      return { stopped: false, error: String(error) };
    } finally {
      set({ sessionStopping: false });
    }
  },

  handleSessionEvent: (event: SessionEvent) => {
    // Update session state based on event
    set((state) => {
      const sessions = state.sessions.map(s =>
        s.sessionId === event.sessionId
          ? { ...s, state: event.type === 'active' ? 'Active' : s.state }
          : s
      );
      return {
        sessions,
        activeSessions: sessions.filter(s => s.state === 'Active'),
      };
    });
  },

  // Plugin actions
  loadPluginRuns: async (sessionId?: string) => {
    try {
      const response = await window.secureClaw.runtime.getPluginRuns({ sessionId });
      set({ pluginRuns: response.runs });
    } catch (error) {
      console.error('Failed to load plugin runs:', error);
    }
  },

  runPlugin: async (runId: string, pluginName: string, sessionId: string, args?: Record<string, unknown>) => {
    try {
      const response = await window.secureClaw.runtime.runPlugin({
        runId,
        pluginName,
        sessionId,
        args,
      });
      
      if (response.queued) {
        await get().loadPluginRuns();
      }
      
      return { queued: response.queued, error: response.error };
    } catch (error) {
      return { queued: false, error: String(error) };
    }
  },

  cancelPluginRun: async (runId: string) => {
    try {
      const response = await window.secureClaw.runtime.cancelPluginRun({ runId });
      
      if (response.cancelled) {
        await get().loadPluginRuns();
      }
      
      return { cancelled: response.cancelled, error: response.error };
    } catch (error) {
      return { cancelled: false, error: String(error) };
    }
  },

  handlePluginEvent: (event: PluginEvent) => {
    // Update plugin run state based on event
    set((state) => ({
      pluginRuns: state.pluginRuns.map(r =>
        r.runId === event.runId
          ? { ...r, state: event.type as any }  // Map event type to state
          : r
      ),
    }));
  },

  // History actions
  loadHistory: async () => {
    set({ historyLoading: true });
    try {
      const filters = get().historyFilters;
      const response = await window.secureClaw.runtime.getHistory(filters);
      set({ history: response.records });
    } catch (error) {
      console.error('Failed to load history:', error);
    } finally {
      set({ historyLoading: false });
    }
  },

  setHistoryFilters: (filters) => {
    set((state) => ({
      historyFilters: { ...state.historyFilters, ...filters },
    }));
  },

  applyHistoryFilters: async () => {
    await get().loadHistory();
  },
}));
```
