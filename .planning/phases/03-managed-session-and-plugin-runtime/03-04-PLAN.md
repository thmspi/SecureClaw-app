Create SessionControl.tsx implementing session start/stop UI with state-aware CTAs and confirmation dialog.

**Implementation structure:**

```typescript
import React, { useState } from 'react';
import { Button } from '../../components/ui/button';
import { Badge } from '../../components/ui/badge';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription, DialogFooter } from '../../components/ui/dialog';
import { useManagementStore } from '../../stores/management-store';

export default function SessionControl() {
  const {
    activeSessions,
    sessionStarting,
    sessionStopping,
    startSession,
    stopSession,
  } = useManagementStore();

  const [showStopDialog, setShowStopDialog] = useState(false);
  const [selectedSessionId, setSelectedSessionId] = useState<string | null>(null);

  const activeSession = activeSessions[0]; // D-03: Multiple sessions allowed, show first for now
  const hasActiveSession = !!activeSession;

  const handleStartSession = async () => {
    const sessionId = crypto.randomUUID();
    const result = await startSession(sessionId);
    
    if (!result.started && result.error) {
      // TODO Phase 4: Show error toast
      console.error('Failed to start session:', result.error);
    }
  };

  const handleStopClick = (sessionId: string) => {
    setSelectedSessionId(sessionId);
    setShowStopDialog(true);
  };

  const handleStopConfirm = async () => {
    if (selectedSessionId) {
      const result = await stopSession(selectedSessionId);
      
      if (!result.stopped && result.error) {
        // TODO Phase 4: Show error toast
        console.error('Failed to stop session:', result.error);
      }
    }
    setShowStopDialog(false);
    setSelectedSessionId(null);
  };

  // Session state badge per UI-SPEC (D-05, D-13)
  const getSessionStateBadge = () => {
    if (sessionStarting) {
      return <Badge variant="secondary">Starting session...</Badge>;
    }
    if (sessionStopping) {
      return <Badge variant="secondary">Stopping session...</Badge>;
    }
    if (hasActiveSession) {
      return <Badge variant="default" className="bg-primary text-primary-foreground">Session active</Badge>;
    }
    return <Badge variant="outline">No active session</Badge>;
  };

  return (
    <div className="space-y-4">
      {/* Status display */}
      <div className="flex items-center justify-between">
        <div className="space-y-1">
          {getSessionStateBadge()}
          {hasActiveSession && activeSession.activeAt && (
            <p className="text-sm text-muted-foreground">
              Active since {new Date(activeSession.activeAt).toLocaleString()}
            </p>
          )}
          {!hasActiveSession && (
            <p className="text-sm text-muted-foreground">
              Start a session to enable plugin operations
            </p>
          )}
        </div>

        {/* Top-right action cluster (D-14) */}
        <div className="flex gap-2">
          {!hasActiveSession && (
            <Button
              onClick={handleStartSession}
              disabled={sessionStarting}
              variant="default"
              size="lg"
            >
              {sessionStarting ? 'Starting...' : 'Start Session'}
            </Button>
          )}

          {hasActiveSession && (
            <Button
              onClick={() => handleStopClick(activeSession.sessionId)}
              disabled={sessionStopping}
              variant="destructive"
              size="lg"
            >
              {sessionStopping ? 'Stopping...' : 'Stop Session'}
            </Button>
          )}
        </div>
      </div>

      {/* Stop confirmation dialog (D-06) */}
      <Dialog open={showStopDialog} onOpenChange={setShowStopDialog}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>Stop Session</DialogTitle>
            <DialogDescription>
              This will terminate the active session and any running plugins. Continue?
            </DialogDescription>
          </DialogHeader>
          <DialogFooter>
            <Button variant="outline" onClick={() => setShowStopDialog(false)}>
              Cancel
            </Button>
            <Button variant="destructive" onClick={handleStopConfirm}>
              Stop Session
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </div>
  );
}
```

Create PluginRuns.tsx implementing plugin list table with lifecycle states and run/retry actions.

**Implementation structure:**

```typescript
import React from 'react';
import { Table, TableHeader, TableRow, TableHead, TableBody, TableCell } from '../../components/ui/table';
import { Badge } from '../../components/ui/badge';
import { Button } from '../../components/ui/button';
import { Tooltip, TooltipContent, TooltipProvider, TooltipTrigger } from '../../components/ui/tooltip';
import { Collapsible, CollapsibleTrigger, CollapsibleContent } from '../../components/ui/collapsible';
import { ChevronDown } from 'lucide-react';
import { useManagementStore } from '../../stores/management-store';
import type { PluginRun } from '../../../shared/runtime/runtime-contracts';

export default function PluginRuns() {
  const { activeSessions, pluginRuns, runPlugin, cancelPluginRun } = useManagementStore();
  
  const hasActiveSession = activeSessions.length > 0;

  const handleRunPlugin = async (pluginName: string) => {
    if (!hasActiveSession) return;
    
    const runId = crypto.randomUUID();
    const sessionId = activeSessions[0].sessionId;
    const result = await runPlugin(runId, pluginName, sessionId);
    
    if (!result.queued && result.error) {
      console.error('Failed to run plugin:', result.error);
    }
  };

  const handleRetry = async (run: PluginRun) => {
    const newRunId = crypto.randomUUID();
    const result = await runPlugin(newRunId, run.pluginName, run.sessionId);
    
    if (!result.queued && result.error) {
      console.error('Failed to retry plugin:', result.error);
    }
  };

  const handleCancel = async (runId: string) => {
    const result = await cancelPluginRun(runId);
    
    if (!result.cancelled && result.error) {
      console.error('Failed to cancel plugin:', result.error);
    }
  };

  // Badge variant per lifecycle state (D-09, UI-SPEC lines 162-167)
  const getStateBadge = (state: PluginRun['state']) => {
    const variants: Record<PluginRun['state'], { variant: any; text: string }> = {
      Idle: { variant: 'outline', text: 'Idle' },
      Queued: { variant: 'secondary', text: 'Queued' },
      Starting: { variant: 'secondary', text: 'Starting' },
      Running: { variant: 'default', text: 'Running' },
      Completed: { variant: 'default', text: 'Completed' },
      Failed: { variant: 'destructive', text: 'Failed' },
    };
    
    const config = variants[state];
    return <Badge variant={config.variant}>{config.text}</Badge>;
  };

  // Empty state when no plugins (UI-SPEC Copywriting)
  if (pluginRuns.length === 0) {
    return (
      <div className="text-center py-8 text-muted-foreground">
        <p className="font-medium">No plugin runs yet</p>
        <p className="text-sm">Start a session, then run a plugin to see execution here.</p>
      </div>
    );
  }

  return (
    <div className="space-y-4">
      {/* Disabled reason when no session (D-07, D-13) */}
      {!hasActiveSession && (
        <p className="text-sm text-muted-foreground">
          Start a session to run plugins
        </p>
      )}

      <Table>
        <TableHeader>
          <TableRow>
            <TableHead>Plugin Name</TableHead>
            <TableHead>Status</TableHead>
            <TableHead>Duration</TableHead>
            <TableHead className="text-right">Actions</TableHead>
          </TableRow>
        </TableHeader>
        <TableBody>
          {pluginRuns.map((run) => (
            <Collapsible key={run.runId}>
              <TableRow>
                <TableCell>
                  <div className="flex items-center gap-2">
                    <CollapsibleTrigger asChild>
                      <Button variant="ghost" size="sm">
                        <ChevronDown className="h-4 w-4" />
                      </Button>
                    </CollapsibleTrigger>
                    {run.pluginName}
                  </div>
                </TableCell>
                <TableCell>{getStateBadge(run.state)}</TableCell>
                <TableCell>
                  {run.durationMs ? `${(run.durationMs / 1000).toFixed(1)}s` : '-'}
                </TableCell>
                <TableCell className="text-right">
                  {/* Retry button for Failed (D-10) */}
                  {run.state === 'Failed' && (
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => handleRetry(run)}
                    >
                      Retry
                    </Button>
                  )}

                  {/* Cancel button for Running */}
                  {(run.state === 'Running' || run.state === 'Starting') && (
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => handleCancel(run.runId)}
                    >
                      Cancel
                    </Button>
                  )}
                </TableCell>
              </TableRow>

              {/* Expandable logs (D-09) */}
              <CollapsibleContent asChild>
                <TableRow>
                  <TableCell colSpan={4}>
                    <div className="p-4 bg-muted rounded-md">
                      <p className="text-sm font-medium mb-2">Logs:</p>
                      <pre className="text-xs font-mono overflow-auto max-h-40">
                        {run.logs.length > 0 ? run.logs.join('\n') : 'No logs available'}
                      </pre>
                      {run.error && (
                        <div className="mt-2 text-sm text-destructive">
                          <strong>Error:</strong> {run.error}
                        </div>
                      )}
                    </div>
                  </TableCell>
                </TableRow>
              </CollapsibleContent>
            </Collapsible>
          ))}
        </TableBody>
      </Table>

      {/* Example: Quick run buttons (hardcoded plugin names for demo) */}
      <div className="flex gap-2 pt-4 border-t">
        <TooltipProvider>
          <Tooltip>
            <TooltipTrigger asChild>
              <span>
                <Button
                  onClick={() => handleRunPlugin('example-plugin')}
                  disabled={!hasActiveSession}
                  size="sm"
                >
                  Run Example Plugin
                </Button>
              </span>
            </TooltipTrigger>
            {!hasActiveSession && (
              <TooltipContent>
                <p>Start a session to run plugins</p>
              </TooltipContent>
            )}
          </Tooltip>
        </TooltipProvider>
      </div>
    </div>
  );
}
```

import { Table, TableHeader, TableRow, TableHead, TableBody, TableCell } from '../../components/ui/table';
import { Select, SelectContent, SelectItem, SelectTrigger, SelectValue } from '../../components/ui/select';
import { Input } from '../../components/ui/input';
import { Badge } from '../../components/ui/badge';
import { Button } from '../../components/ui/button';
import { Collapsible, CollapsibleTrigger, CollapsibleContent } from '../../components/ui/collapsible';
import { ChevronDown } from 'lucide-react';
import { useManagementStore } from '../../stores/management-store';
import type { RuntimeHistoryRecord } from '../../../shared/runtime/runtime-contracts';

export default function OperationHistory() {
  const { history, historyFilters, historyLoading, setHistoryFilters, applyHistoryFilters } = useManagementStore();

  const handleFilterChange = (key: string, value: string | undefined) => {
    setHistoryFilters({ [key]: value || undefined });
  };

  const handleApplyFilters = () => {
    void applyHistoryFilters();
  };

  // Badge variant per operation type and status (D-16, UI-SPEC lines 179-180)
  const getTypeBadge = (type: RuntimeHistoryRecord['operationType']) => {
    const variants = {
      session_start: { variant: 'secondary' as const, text: 'Session Start' },
      session_stop: { variant: 'secondary' as const, text: 'Session Stop' },
      plugin_run: { variant: 'outline' as const, text: 'Plugin Run' },
    };
    const config = variants[type];
    return <Badge variant={config.variant}>{config.text}</Badge>;
  };

  const getStatusBadge = (status: RuntimeHistoryRecord['status']) => {
    const variants = {
      success: { variant: 'default' as const, text: 'Success' },
      failed: { variant: 'destructive' as const, text: 'Failed' },
      cancelled: { variant: 'secondary' as const, text: 'Cancelled' },
    };
    const config = variants[status];
    return <Badge variant={config.variant}>{config.text}</Badge>;
  };

  // Empty state when no history
  if (history.length === 0 && !historyLoading) {
    return (
      <div className="text-center py-8 text-muted-foreground">
        <p className="font-medium">No operations recorded</p>
        <p className="text-sm">Session and plugin operations will appear here.</p>
      </div>
    );
  }

  return (
    <div className="space-y-4">
      {/* Filter controls (D-18) */}
      <div className="flex gap-4 items-end">
        <div className="flex-1">
          <label className="text-sm font-medium mb-1 block">Operation Type</label>
          <Select
            value={historyFilters.operationType || 'all'}
            onValueChange={(val) => handleFilterChange('operationType', val === 'all' ? undefined : val)}
          >
            <SelectTrigger>
              <SelectValue placeholder="All" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">All</SelectItem>
              <SelectItem value="session_start">Session Start</SelectItem>
              <SelectItem value="session_stop">Session Stop</SelectItem>
              <SelectItem value="plugin_run">Plugin Run</SelectItem>
            </SelectContent>
          </Select>
        </div>

        <div className="flex-1">
          <label className="text-sm font-medium mb-1 block">Status</label>
          <Select
            value={historyFilters.status || 'all'}
            onValueChange={(val) => handleFilterChange('status', val === 'all' ? undefined : val)}
          >
            <SelectTrigger>
              <SelectValue placeholder="All" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">All</SelectItem>
              <SelectItem value="success">Success</SelectItem>
              <SelectItem value="failed">Failed</SelectItem>
              <SelectItem value="cancelled">Cancelled</SelectItem>
            </SelectContent>
          </Select>
        </div>

        <div className="flex-1">
          <label className="text-sm font-medium mb-1 block">From Date</label>
          <Input
            type="date"
            value={historyFilters.fromDate || ''}
            onChange={(e) => handleFilterChange('fromDate', e.target.value)}
          />
        </div>

        <div className="flex-1">
          <label className="text-sm font-medium mb-1 block">To Date</label>
          <Input
            type="date"
            value={historyFilters.toDate || ''}
            onChange={(e) => handleFilterChange('toDate', e.target.value)}
          />
        </div>

        <Button onClick={handleApplyFilters} disabled={historyLoading}>
          {historyLoading ? 'Loading...' : 'Apply Filters'}
        </Button>
      </div>

      {/* History table */}
      <Table>
        <TableHeader>
          <TableRow>
            <TableHead>Timestamp</TableHead>
            <TableHead>Operation</TableHead>
            <TableHead>Target</TableHead>
            <TableHead>Status</TableHead>
            <TableHead>Duration</TableHead>
          </TableRow>
        </TableHeader>
        <TableBody>
          {history.map((record) => (
            <Collapsible key={record.id}>
              <TableRow>
                <TableCell>
                  <div className="flex items-center gap-2">
                    <CollapsibleTrigger asChild>
                      <Button variant="ghost" size="sm">
                        <ChevronDown className="h-4 w-4" />
                      </Button>
                    </CollapsibleTrigger>
                    {new Date(record.startedAt).toLocaleString()}
                  </div>
                </TableCell>
                <TableCell>{getTypeBadge(record.operationType)}</TableCell>
                <TableCell>{record.targetName || '-'}</TableCell>
                <TableCell>{getStatusBadge(record.status)}</TableCell>
                <TableCell>
                  {record.durationMs ? `${(record.durationMs / 1000).toFixed(1)}s` : '-'}
                </TableCell>
              </TableRow>

              {/* Expandable details (D-16) */}
              <CollapsibleContent asChild>
                <TableRow>
                  <TableCell colSpan={5}>
                    <div className="p-4 bg-muted rounded-md space-y-2">
                      <div>
                        <strong className="text-sm">ID:</strong> <span className="text-sm font-mono">{record.id}</span>
                      </div>
                      {record.errorMessage && (
                        <div className="text-destructive">
                          <strong className="text-sm">Error:</strong> <span className="text-sm">{record.errorMessage}</span>
                        </div>
                      )}
                      {record.errorDetails && (
                        <pre className="text-xs font-mono overflow-auto max-h-40 mt-2">
                          {record.errorDetails}
                        </pre>
                      )}
                      {record.metadata && (
                        <div>
                          <strong className="text-sm">Metadata:</strong>
                          <pre className="text-xs font-mono mt-1">
                            {JSON.stringify(record.metadata, null, 2)}
                          </pre>
                        </div>
                      )}
                    </div>
                  </TableCell>
                </TableRow>
              </CollapsibleContent>
            </Collapsible>
          ))}
        </TableBody>
      </Table>
    </div>
  );
}
