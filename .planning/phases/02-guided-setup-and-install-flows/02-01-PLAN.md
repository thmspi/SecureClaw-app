2. `src/main/install/prerequisite-service.ts`:
   - Import spawn from 'child_process', statfsSync from 'fs', mkdirSync/rmdirSync from 'fs', app/net from 'electron', join from 'path'
   - Import PrerequisiteCheck, PrerequisiteResult from shared/install
   
   Helper function `collectOutput(proc)`: Returns promise that resolves stdout/stderr combined on exit code 0
   
   Helper function `semverGte(a, b)`: Compare semantic versions, returns true if a >= b
   
   async function `checkNodeVersion()`: 
   - Spawn 'node', ['--version']
   - Parse version from output (strip 'v' prefix)
   - Return PrerequisiteCheck with id='node-version', status based on >= 18.0.0
   - On error return failed with message "Not found. Install from nodejs.org"
   
   async function `checkPython()`:
   - Try 'python3' first, then 'python'
   - Parse version from 'Python X.Y.Z' output
   - Return PrerequisiteCheck with id='python', status based on >= 3.10.0
   - On all failures return failed with install instructions
   
   function `checkDiskSpace()`: (synchronous)
   - Use statfsSync on app.getPath('userData')
   - Calculate free GB: (stats.bfree * stats.bsize) / (1024**3)
   - Return PrerequisiteCheck with id='disk-space', status based on >= 5GB
   
   function `checkWritePermissions()`: (synchronous)
   - Create test dir at join(app.getPath('userData'), '.perm-test')
   - Remove it immediately
   - Return passed on success, failed on error with id='permissions'
   
   async function `checkNetwork()`:
   - Use net.isOnline()
   - Return PrerequisiteCheck with id='network', status 'passed' or 'warning'
   
   export async function `runAllPrerequisiteChecks()`:
   - Run all checks in parallel with Promise.all
   - Return { allPassed: all passed or warning, checks }

2. `src/main/install/install-state-service.ts`:
   - Import Database from 'better-sqlite3'
   - Import app from 'electron'
   - Import join from 'path'
   - Import InstallState, InstallTarget, InstallStatus from shared/install
   
   const DB_PATH = join(app.getPath('userData'), 'secureclaw.db')
   
   let db: Database.Database | null = null
   
   function `getDb()`:
   - If db null, create new Database(DB_PATH)
   - Set pragma journal_mode = WAL
   - Create install_state table if not exists with columns:
     - id TEXT PRIMARY KEY DEFAULT 'singleton'
     - target TEXT NOT NULL
     - status TEXT NOT NULL
     - current_step INTEGER NOT NULL DEFAULT 0
     - total_steps INTEGER NOT NULL DEFAULT 5
     - step_name TEXT
     - error_message TEXT
     - error_details TEXT
     - started_at TEXT
     - updated_at TEXT NOT NULL
     - completed_steps TEXT DEFAULT '[]'
   - Return db
   
   export function `saveInstallState(state: InstallState)`:
   - Use INSERT ... ON CONFLICT DO UPDATE pattern
   - Serialize completedSteps as JSON.stringify
   
   export function `loadInstallState()`: InstallState | null
   - SELECT where id = 'singleton'
   - Return null if no row
   - Parse completed_steps from JSON, map column names to camelCase
   
   export function `clearInstallState()`:
   - DELETE where id = 'singleton'
   
   // For testing - allows injecting in-memory db
   export function `_setDbForTesting(testDb: Database.Database)`:
   - Set db = testDb
